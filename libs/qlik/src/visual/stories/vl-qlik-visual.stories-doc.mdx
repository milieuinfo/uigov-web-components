# Visual

<VluxMetaData id="qlik-qlik-visual" />

Gebruik de <code>vl-qlik-visual</code> component om een Qlik visual in een webpagina te renderen, zonder het gebruik
van een iFrame.


## Voorbeeld

```js
import "@domg-wc/qlik/visual";
import {Qlik, STARDUST} from "@domg/qlik-lib";

const connection = new Qlik("omgevingsloketrapport.omgeving.vlaanderen.be", "594c04b6-f319-4cb9-962c-24b0f7aa6f5e");
await connection.init();
this.stardust = await STARDUST(connection.app);
```

```html
<vl-qlik-visual
    qlik-id="RXjdD"
    type="kpi"
    height="200px"
    width="200px"
    .stardust="${stardust}"
></vl-qlik-visual>
```

## Types

export type Filter = {
    id: string,
    filter: FilterObj
}

type FilterObj = {
    name: string,

    field: string,
    nrOfValues: number
}

export type DashboardVisualization = {
    type: string,
    id: string,
    height: string,
    width?: string,
    label: string,
    "align-label"?: string,
    "align-visual"?: string,
    colSize?: number,
    properties?: object,
    options?: object
}

export type InfoblockVisualization = {
    title: string,
    icon: string,
    type: string,
    id: string,
    height: string,

    align: string,
    colSize?: number,
    properties?: object

}

export type Views = {
    [key: string]: {
        order: number
        label: string,
        visualisations: DashboardVisualization[][]
    },
}
